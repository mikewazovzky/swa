Создание проекта laravel5

$ composer create-project laravel/laravel laravel5 "5.3.*"
$ composer require "laravelcollective/html":"^5.3.0"
добавить ServiceProvider и 'aliases' в config\app.php
$ php artisan m ake:auth
параметры базы данных в переменных среды (.env)
$ php artisan migrate
указать в качестве папки домена laravel5/public
